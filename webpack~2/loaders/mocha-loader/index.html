mocha-loader

Install

npm install --save-dev mocha-loader

Usage

Configuration (recommended)

webpack.config.js

module.exports = {
  entry: './entry.js',
  output: {
    path: __dirname,
    filename: 'bundle.js'
  }
}

import test from './test'

CLI

webpack --module-bind 'mocha-loader!./test'

import test from './test'

Require

import test from 'mocha-loader!./test'
