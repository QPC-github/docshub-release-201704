function filterProjectInfo($info, $additional_whitelist = array()) {
  $whitelist = array(
    '_info_file_ctime',
    'datestamp',
    'major',
    'name',
    'package',
    'project',
    'project status url',
    'version',
  );
  $whitelist = array_merge($whitelist, $additional_whitelist);
  return array_intersect_key($info, array_combine($whitelist, $whitelist));
}
