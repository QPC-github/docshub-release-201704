protected function hashBase64($data) {
  $hash = base64_encode(hash('sha256', $data, TRUE));
  // Modify the hash so it's safe to use in PostgreSQL identifiers.
  return strtr($hash, array('+' => '_', '/' => '_', '=' => ''));
}
