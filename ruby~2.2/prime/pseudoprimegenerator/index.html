# File lib/prime.rb, line 275
def each
  return self.dup unless block_given?
  if @ubound
    last_value = nil
    loop do
      prime = succ
      break last_value if prime > @ubound
      last_value = yield prime
    end
  else
    loop do
      yield succ
    end
  end
end
