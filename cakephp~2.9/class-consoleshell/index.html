$_methodPatternssource

_methodPatterns

array(
    'help' => '/^(help|\?)/',
    '_exit' => '/^(quit|exit)/',
    '_models' => '/^models/i',
    '_bind' => '/^(\w+) bind (\w+) (\w+)/',
    '_unbind' => '/^(\w+) unbind (\w+) (\w+)/',
    '_find' => '/.+->find/',
    '_save' => '/.+->save/',
    '_columns' => '/^(\w+) columns/',
    '_routesReload' => '/^routes\s+reload/i',
    '_routesShow' => '/^routes\s+show/i',
    '_routeToString' => '/^route\s+(\(.*\))$/i',
    '_routeToArray' => '/^route\s+(.*)$/i',
)
